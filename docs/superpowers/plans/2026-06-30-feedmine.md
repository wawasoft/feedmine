# Feedmine Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an iOS 18+ iPhone RSS reader with a Tinder-style swipe feed, curated default feeds, and on-device personalization.

**Architecture:** SwiftUI app with three tabs (Sources, Feed, Favorites). SwiftData for persistence, FeedKit for RSS/Atom parsing, and a simple Personalizer struct for swipe-based content scoring. All services are isolated — RSSEngine is an actor, FeedStore is an @Observable class injected via the SwiftUI environment.

**Tech Stack:** SwiftUI, SwiftData, FeedKit (SPM), iOS 18+, Swift Testing framework

## Global Constraints

- iOS 18+ deployment target (iPhone only)
- Swift 6 strict concurrency (use `actor` for network I/O, `@Observable` for view state)
- All persistence via SwiftData `@Model` classes except seen-set (UserDefaults)
- One external dependency: FeedKit (MIT, via SPM)
- TDD for services (RSSEngine, Personalizer, FeedStore); manual testing for views
- Dark mode support (use semantic colors, no hardcoded colors)
- Minimum 15 curated feeds across 5 categories shipped as bundled JSON

- [ ] **Step 5: Write RSSEngine.swift**

```swift
import Foundation
import FeedKit

enum RSSEngineError: Error, Equatable {
    case invalidURL
    case parseFailed
    case timeout
}

actor RSSEngine {

    // MARK: - Public types

    struct ParsedFeed {
        let title: String
        let iconURL: String?
    }

    struct PostData {
        let guid: String
        let title: String
        let excerpt: String
        let contentHTML: String
        let url: String
        let imageURL: String?
        let publishDate: Date
    }

    // MARK: - Public API

    func fetch(url: URL) async throws -> (ParsedFeed, [PostData]) {
        let parser = FeedParser(URL: url)

        return try await withCheckedThrowingContinuation { continuation in
            parser.parseAsync(queue: .global(qos: .userInitiated)) { [weak self] result in
                guard let self else { return }
                switch result {
                case .success(let feed):
                    let output = self.extract(feed: feed)
                    continuation.resume(returning: output)
                case .failure:
                    continuation.resume(throwing: RSSEngineError.parseFailed)
                }
            }
        }
    }

    // MARK: - Private

    private func extract(feed: FeedKit.Feed) -> (ParsedFeed, [PostData]) {
        switch feed {
        case .atom(let atom):
            return extractAtom(atom)
        case .rss(let rss):
            return extractRSS(rss)
        case .json(let json):
            return extractJSON(json)
        }
    }

    private func extractRSS(_ rss: RSSFeed) -> (ParsedFeed, [PostData]) {
        let parsed = ParsedFeed(
            title: rss.title ?? "Untitled",
            iconURL: rss.image?.url
        )

        let posts: [PostData] = (rss.items ?? []).compactMap { item in
            guard let title = item.title,
                  let guid = item.guid?.value ?? item.link else { return nil }

            return PostData(
                guid: guid,
                title: title,
                excerpt: stripHTML(item.description ?? "").truncated(to: 200),
                contentHTML: item.description ?? "",
                url: item.link ?? "",
                imageURL: item.enclosure?.attributes?.url ?? extractImageFromHTML(item.description ?? ""),
                publishDate: item.pubDate ?? Date()
            )
        }

        return (parsed, posts)
    }

    private func extractAtom(_ atom: AtomFeed) -> (ParsedFeed, [PostData]) {
        let parsed = ParsedFeed(
            title: atom.title ?? "Untitled",
            iconURL: atom.icon ?? atom.logo
        )

        let posts: [PostData] = (atom.entries ?? []).compactMap { entry in
            guard let title = entry.title,
                  let guid = entry.id else { return nil }

            return PostData(
                guid: guid,
                title: title,
                excerpt: stripHTML(entry.summary?.value ?? "").truncated(to: 200),
                contentHTML: entry.content?.value ?? entry.summary?.value ?? "",
                url: entry.links?.first?.attributes?.href ?? "",
                imageURL: extractImageFromHTML(entry.content?.value ?? ""),
                publishDate: entry.published ?? entry.updated ?? Date()
            )
        }

        return (parsed, posts)
    }

    private func extractJSON(_ json: JSONFeed) -> (ParsedFeed, [PostData]) {
        let parsed = ParsedFeed(
            title: json.title ?? "Untitled",
            iconURL: json.icon ?? json.favicon
        )

        let posts: [PostData] = (json.items ?? []).compactMap { item in
            guard let title = item.title,
                  let guid = item.id,
                  let url = item.url else { return nil }

            return PostData(
                guid: guid,
                title: title,
                excerpt: stripHTML(item.contentText ?? item.contentHtml ?? "").truncated(to: 200),
                contentHTML: item.contentHtml ?? item.contentText ?? "",
                url: url,
                imageURL: item.image ?? item.bannerImage ?? extractImageFromHTML(item.contentHtml ?? ""),
                publishDate: item.datePublished ?? Date()
            )
        }

        return (parsed, posts)
    }

    private func stripHTML(_ html: String) -> String {
        guard let data = html.data(using: .utf8) else { return html }
        if let plain = try? NSAttributedString(
            data: data,
            options: [.documentType: NSAttributedString.DocumentType.html],
            documentAttributes: nil
        ).string {
            return plain
        }
        return html
    }

    private func extractImageFromHTML(_ html: String) -> String? {
        guard !html.isEmpty else { return nil }
        // Simple regex for <img src="...">
        let pattern = #"<img[^>]+src="([^"]+)""#
        guard let regex = try? NSRegularExpression(pattern: pattern),
              let match = regex.firstMatch(in: html, range: NSRange(html.startIndex..., in: html)) else {
            return nil
        }
        if let range = Range(match.range(at: 1), in: html) {
            return String(html[range])
        }
        return nil
    }
}

// MARK: - Helpers

extension String {
    func truncated(to maxLength: Int) -> String {
        count <= maxLength ? self : String(prefix(maxLength)) + "…"
    }
}
```

```swift
import Foundation

struct Personalizer {

    /// Returns a relevance score from 0.0 (disliked) to 1.0 (liked).
    /// 0.5 means neutral / no data.
    func score(category: String, feedURL: String, events: [SwipeEvent]) -> Double {
        let now = Date()

        let categoryEvents = events.filter { $0.feedCategory == category }
        let sourceEvents = events.filter { $0.feedURL == feedURL }

        let categoryBoost = boost(from: categoryEvents, now: now)
        let sourceBoost = boost(from: sourceEvents, now: now)

        return (categoryBoost * 0.4) + (sourceBoost * 0.6)
    }

    /// Reorders posts so higher-scored ones tend to appear first.
    /// Uses weighted randomness — no guarantee, just tendency.
    func reorderQueue(_ posts: [Post], events: [SwipeEvent]) -> [Post] {
        posts.sorted { a, b in
            let scoreA = score(category: a.feedCategory, feedURL: a.url, events: events)
            let scoreB = score(category: b.feedCategory, feedURL: b.url, events: events)

            // Add jitter (±25%) to prevent strict ordering
            let randomA = scoreA * Double.random(in: 0.75...1.25)
            let randomB = scoreB * Double.random(in: 0.75...1.25)

            return randomA > randomB
        }
    }

    // MARK: - Private

    private func boost(from events: [SwipeEvent], now: Date) -> Double {
        guard !events.isEmpty else { return 0.5 }

        var weightedLikes = 0.0
        var weightedDislikes = 0.0

        for event in events {
            let weight = decayWeight(event.timestamp, now: now)
            if event.action == "like" {
                weightedLikes += weight
            } else {
                weightedDislikes += weight
            }
        }

        let total = weightedLikes + weightedDislikes
        guard total > 0 else { return 0.5 }
        return weightedLikes / total
    }

    private func decayWeight(_ timestamp: Date, now: Date) -> Double {
        let days = now.timeIntervalSince(timestamp) / (24 * 3600)
        if days > 60 { return 0.0 }
        if days > 30 { return 0.5 }
        return 1.0
    }
}
```

```swift
import Foundation
import SwiftData
import Observation

enum SwipeAction {
    case like
    case dislike
}

enum FeedStoreError: Error, Equatable {
    case invalidURL
    case feedAlreadyExists
}

@Observable
final class FeedStore {
    // MARK: - Published state

    private(set) var isLoading = false
    private(set) var isOffline = false

    // MARK: - Dependencies

    private let modelContext: ModelContext
    private let engine: any RSSEngineProtocol
    private let personalizer = Personalizer()
    private var seenSets: [String: Set<String>] = [:]

    // MARK: - Init

    init(modelContext: ModelContext, engine: any RSSEngineProtocol = RSSEngine()) {
        self.modelContext = modelContext
        self.engine = engine
        loadSeenSets()
    }

    // MARK: - Feed management

    var enabledFeeds: [Feed] {
        let descriptor = FetchDescriptor<Feed>(predicate: #Predicate { $0.isEnabled == true })
        return (try? modelContext.fetch(descriptor)) ?? []
    }

    var allFeeds: [Feed] {
        (try? modelContext.fetch(FetchDescriptor<Feed>())) ?? []
    }

    func seedCuratedFeedsIfNeeded() {
        let existing = (try? modelContext.fetch(FetchDescriptor<Feed>())) ?? []
        guard existing.isEmpty else { return }

        guard let url = Bundle.main.url(forResource: "curated_feeds", withExtension: "json"),
              let data = try? Data(contentsOf: url),
              let entries = try? JSONDecoder().decode([CuratedFeedEntry].self, from: data) else {
            print("⚠️ Failed to load curated_feeds.json")
            return
        }

        for entry in entries {
            let feed = Feed(
                url: entry.url,
                title: entry.title,
                iconURL: entry.iconURL,
                category: entry.category,
                isCurated: true,
                isEnabled: true
            )
            modelContext.insert(feed)
        }

        try? modelContext.save()
    }

    func addFeed(url rawURL: String) async throws -> Feed {
        guard let url = URL(string: rawURL),
              url.scheme == "http" || url.scheme == "https" else {
            throw FeedStoreError.invalidURL
        }

        // Check for duplicates
        let descriptor = FetchDescriptor<Feed>(predicate: #Predicate { $0.url == rawURL })
        if let existing = try? modelContext.fetch(descriptor).first {
            return existing
        }

        let (parsed, posts) = try await engine.fetch(url: url)

        let feed = Feed(
            url: rawURL,
            title: parsed.title,
            iconURL: parsed.iconURL,
            category: "General",
            isCurated: false,
            isEnabled: true
        )
        feed.lastFetched = Date()
        modelContext.insert(feed)

        for postData in posts {
            let post = Post(
                guid: postData.guid,
                title: postData.title,
                excerpt: postData.excerpt,
                contentHTML: postData.contentHTML,
                url: postData.url,
                imageURL: postData.imageURL,
                publishDate: postData.publishDate,
                feedTitle: parsed.title,
                feedURL: rawURL,
                feedIconURL: parsed.iconURL,
                feedCategory: "General"
            )
            feed.posts.append(post)
        }

        try? modelContext.save()
        return feed
    }

    func removeFeed(_ feed: Feed) {
        modelContext.delete(feed)
        try? modelContext.save()
    }

    func toggleFeed(_ feed: Feed, isEnabled: Bool) {
        feed.isEnabled = isEnabled
        try? modelContext.save()
    }

    // MARK: - Card queue

    func nextBatch(count: Int) async -> [Post] {
        let feeds = enabledFeeds
        guard !feeds.isEmpty else { return [] }

        isLoading = true
        defer { isLoading = false }

        var allPosts: [Post] = []

        for feed in feeds {
            // Fetch if stale or no cached posts
            if feed.lastFetched == nil || Date().timeIntervalSince(feed.lastFetched!) > 900 {
                await refreshFeed(feed)
            }
            // Collect unseen posts from this feed
            let seen = seenGUIDs(for: feed.url)
            let newPosts = feed.posts.filter { !seen.contains($0.guid) }
            allPosts.append(contentsOf: newPosts)
        }

        // Deduplicate by guid (in case same post appears in multiple feeds)
        var seenGuids = Set<String>()
        allPosts = allPosts.filter { seenGuids.insert($0.guid).inserted }

        // Sort by publish date descending
        allPosts.sort { $0.publishDate > $1.publishDate }

        // Shuffle posts within the same minute to avoid clustering
        allPosts = shuffleSameMinutePosts(allPosts)

        // Apply personalization
        let events = (try? modelContext.fetch(FetchDescriptor<SwipeEvent>())) ?? []
        allPosts = personalizer.reorderQueue(allPosts, events: events)

        return Array(allPosts.prefix(count))
    }

    func markSeen(_ post: Post) {
        var seen = seenGUIDs(for: post.feedURL)
        seen.insert(post.guid)
        saveSeenGUIDs(seen, for: post.feedURL)
    }

    // MARK: - Favorites

    func likePost(_ post: Post) {
        let saved = SavedPost(
            postGuid: post.guid,
            title: post.title,
            excerpt: post.excerpt,
            url: post.url,
            imageURL: post.imageURL,
            feedTitle: post.feedTitle,
            feedCategory: post.feedCategory
        )
        modelContext.insert(saved)
        try? modelContext.save()
    }

    func getFavorites() -> [SavedPost] {
        var descriptor = FetchDescriptor<SavedPost>(sortBy: [SortDescriptor(\.savedAt, order: .reverse)])
        return (try? modelContext.fetch(descriptor)) ?? []
    }

    func removeFavorite(_ savedPost: SavedPost) {
        modelContext.delete(savedPost)
        try? modelContext.save()
    }

    // MARK: - Swipe tracking

    func recordSwipe(post: Post, action: SwipeAction) {
        let actionString = action == .like ? "like" : "dislike"
        let event = SwipeEvent(
            postGuid: post.guid,
            feedURL: post.feedURL,
            feedCategory: post.feedCategory,
            action: actionString
        )
        modelContext.insert(event)
        try? modelContext.save()
    }

    // MARK: - Private

    private func refreshFeed(_ feed: Feed) async {
        guard let url = URL(string: feed.url) else { return }
        do {
            let (parsed, postDataList) = try await engine.fetch(url: url)
            feed.title = parsed.title
            feed.iconURL = parsed.iconURL
            feed.lastFetched = Date()

            for postData in postDataList {
                // Skip duplicates
                if feed.posts.contains(where: { $0.guid == postData.guid }) { continue }
                let post = Post(
                    guid: postData.guid,
                    title: postData.title,
                    excerpt: postData.excerpt,
                    contentHTML: postData.contentHTML,
                    url: postData.url,
                    imageURL: postData.imageURL,
                    publishDate: postData.publishDate,
                    feedTitle: parsed.title,
                    feedURL: feed.url,
                    feedIconURL: parsed.iconURL,
                    feedCategory: feed.category
                )
                feed.posts.append(post)
            }
            try? modelContext.save()
            isOffline = false
        } catch {
            isOffline = true
            print("⚠️ Failed to refresh feed \(feed.title): \(error)")
        }
    }

    private func shuffleSameMinutePosts(_ posts: [Post]) -> [Post] {
        guard posts.count > 1 else { return posts }
        var result: [Post] = []
        var cluster: [Post] = []

        for (i, post) in posts.enumerated() {
            if let last = cluster.last,
               abs(post.publishDate.timeIntervalSince(last.publishDate)) <= 60 {
                cluster.append(post)
            } else {
                result.append(contentsOf: cluster.shuffled())
                cluster = [post]
            }

            if i == posts.count - 1 {
                result.append(contentsOf: cluster.shuffled())
            }
        }

        return result
    }

    // MARK: - Seen-set persistence

    private func seenGUIDs(for feedURL: String) -> Set<String> {
        seenSets[feedURL] ?? []
    }

    private func saveSeenGUIDs(_ guids: Set<String>, for feedURL: String) {
        let trimmed = Array(guids).suffix(500)
        seenSets[feedURL] = Set(trimmed)
        persistSeenSets()
    }

    private func loadSeenSets() {
        guard let data = UserDefaults.standard.data(forKey: "feedmine_seen_sets"),
              let decoded = try? JSONDecoder().decode([String: [String]].self, from: data) else {
            return
        }
        for (key, guids) in decoded {
            seenSets[key] = Set(guids.suffix(500))
        }
    }

    private func persistSeenSets() {
        let dict = seenSets.mapValues { Array($0.suffix(500)) }
        if let data = try? JSONEncoder().encode(dict) {
            UserDefaults.standard.set(data, forKey: "feedmine_seen_sets")
        }
    }
}

// MARK: - Curated feed JSON shape

private struct CuratedFeedEntry: Decodable {
    let title: String
    let url: String
    let category: String
    let iconURL: String?
}
```

```swift
import SwiftUI

struct SwipeOverlay: View {
    let offset: CGSize

    private let screenWidth: CGFloat = UIScreen.main.bounds.width
    private let threshold: CGFloat { screenWidth * 0.4 }

    var body: some View {
        HStack {
            // Left side: NOPE
            if offset.width < 0 {
                stamp(
                    text: "NOPE",
                    color: .red,
                    opacity: min(abs(offset.width) / threshold, 1.0)
                )
            }

            Spacer()

            // Right side: LIKE
            if offset.width > 0 {
                stamp(
                    text: "LIKE",
                    color: .green,
                    opacity: min(offset.width / threshold, 1.0)
                )
            }
        }
        .padding(.horizontal, 40)
        .allowsHitTesting(false)
    }

    private func stamp(text: String, color: Color, opacity: Double) -> some View {
        Text(text)
            .font(.system(size: 48, weight: .heavy, design: .rounded))
            .foregroundColor(color)
            .padding(.horizontal, 16)
            .padding(.vertical, 8)
```

```swift
import SwiftUI

struct FeedView: View {
    @Environment(FeedStore.self) private var store

    @State private var cards: [Post] = []
    @State private var currentIndex = 0
    @State private var dragOffset = CGSize.zero
    @State private var selectedURL: URL?

    private let screenWidth: CGFloat = UIScreen.main.bounds.width
    private let swipeThreshold: CGFloat { screenWidth * 0.4 }

    var body: some View {
        NavigationStack {
            ZStack {
                if cards.isEmpty {
                    emptyState
                } else {
                    cardStack
                }
            }
            .navigationTitle("Feedmine")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    if store.isOffline {
                        Label("Offline", systemImage: "wifi.slash")
                            .font(.caption)
                            .foregroundStyle(.orange)
                    }
                }
            }
            .task { await loadInitialBatch() }
            .sheet(item: $selectedURL) { url in
                SafariView(url: url)
            }
        }
    }

    // MARK: - Card stack

    @ViewBuilder
    private var cardStack: some View {
        let visibleCards = cards.dropFirst(currentIndex).prefix(5)

        ZStack {
            ForEach(Array(visibleCards.enumerated()), id: \.element.guid) { index, post in
                let isTop = index == 0

                FeedCardView(post: post)
                    .frame(maxWidth: .infinity, maxHeight: .infinity, alignment: .top)
                    .padding(.horizontal, 16)
                    .padding(.top, 8)
                    .zIndex(Double(5 - index))
                    .offset(isTop ? dragOffset : .zero)
                    .rotationEffect(isTop ? .degrees(Double(dragOffset.width / 20)) : .zero)
                    .scaleEffect(isTop ? 1.0 : 1.0 - Double(index) * 0.03)
                    .offset(y: CGFloat(index) * 12)
                    .opacity(isTop && abs(dragOffset.width) > swipeThreshold ? 0 : 1.0)
                    .overlay {
                        if isTop { SwipeOverlay(offset: dragOffset) }
                    }
                    .gesture(isTop ? swipeGesture : nil)
                    .onTapGesture {
                        if isTop, let url = URL(string: post.url) {
                            selectedURL = url
                        }
                    }
                    .allowsHitTesting(isTop)
            }
        }
        .onChange(of: currentIndex) { _, _ in
            preloadIfNeeded()
        }
    }

    private var swipeGesture: some Gesture {
        DragGesture()
            .onChanged { value in
                dragOffset = value.translation
            }
            .onEnded { value in
                let width = value.translation.width
                if abs(width) > swipeThreshold {
                    commitSwipe(liked: width > 0)
                } else {
                    withAnimation(.spring(bounce: 0.6)) {
                        dragOffset = .zero
                    }
                }
            }
    }

    // MARK: - Empty state

    private var emptyState: some View {
        ContentUnavailableView(
            "Add Some Feeds",
            systemImage: "antenna.radiowaves.left.and.right",
            description: Text("Head to the Sources tab to enable some feeds and get started.")
        )
    }

    // MARK: - Logic

    private func loadInitialBatch() async {
        let batch = await store.nextBatch(count: 10)
        withAnimation {
            cards = batch
            currentIndex = 0
        }
    }

    private func commitSwipe(liked: Bool) {
        guard currentIndex < cards.count else { return }
        let post = cards[currentIndex]
        let cardWidth: CGFloat = screenWidth - 32

        // Record the swipe
        store.recordSwipe(post: post, action: liked ? .like : .dislike)
        if liked { store.likePost(post) }

        // Animate card off screen
        let finalX: CGFloat = liked ? screenWidth * 1.5 : -screenWidth * 1.5

        withAnimation(.spring(response: 0.4, dampingFraction: 0.7)) {
            dragOffset = CGSize(width: finalX, height: dragOffset.height)
        }

        // After animation, advance
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.35) {
            store.markSeen(post)
            dragOffset = .zero
            currentIndex += 1
        }
    }

    private func preloadIfNeeded() {
        let remaining = cards.count - currentIndex
        guard remaining < 5 else { return }

        Task {
            let batch = await store.nextBatch(count: 10)
            guard !batch.isEmpty else { return }
            withAnimation {
                cards.append(contentsOf: batch)
            }
        }
    }
}

extension URL: @retroactive Identifiable {
    public var id: String { absoluteString }
}
```

```swift
                        Color.secondary.opacity(0.2)
                    }
                }
                .frame(width: 60, height: 60)
                .clipShape(RoundedRectangle(cornerRadius: 8))
            }

            VStack(alignment: .leading, spacing: 4) {

                    .font(.subheadline.weight(.medium))
                    .lineLimit(2)

                        .font(.caption2.weight(.medium))

                    Text("·")
                        .foregroundStyle(.tertiary)

                    Text(saved.feedTitle)
                        .font(.caption2)
                        .foregroundStyle(.secondary)
                }
            }
        }
        .contentShape(Rectangle())
        .onTapGesture {
            if let url = URL(string: saved.url) {

            }
        }
```
